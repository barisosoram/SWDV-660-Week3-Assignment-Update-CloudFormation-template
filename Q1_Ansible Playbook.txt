---
# tasks file for helloworld

- hosts: webservers
  tasks:
    - name: upgrade all packages
      become: yes
      yum: name=* state=latest

    - name: Installing node, npm and mongodb
      become: yes
      yum:
         enablerepo: epel
         state: installed
         name:
         - nodejs
         - npm
		 - mongodb

    - name: Copying the application file
      become: yes
      copy:
        src: /home/ec2-user/SWDV660/helloworld.js
        dest: /home/ec2-user/
        owner: ec2-user
        group: ec2-user
        mode: 0644

    - name: Copying the upstart file
      become: yes
      copy:
        src: /home/ec2-user/SWDV660/helloworld.conf
        dest: /etc/init/helloworld.conf                                   
        owner: root
        group: root
        mode: 0644

    - name: restart helloworld
      service: name=helloworld state=restarted
      become: yes
